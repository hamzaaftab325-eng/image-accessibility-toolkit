     6→// Dynamically import ZAI, supporting global installation (Node.js and Bun)
     7→async function getZAI() {
     8→    try {
     9→        // Try importing from package name (when globally installed)
    10→        const packageName = 'z-ai-web-dev-sdk';
    11→        const { default: ZAI } = await import(packageName);
    12→        return ZAI;
    13→    }
    14→    catch (error1) {
    15→        try {
    16→            // Try importing directly from global node_modules (container environment fix)
    17→            const globalPackagePath = '/usr/lib/node_modules/z-ai-web-dev-sdk/dist/index.js';
    18→            const { default: ZAI } = await import(globalPackagePath);
    19→            return ZAI;
    20→        }
    21→        catch (error2) {
    22→            try {
    23→                // Dynamically get global path (supporting Node.js and Bun)
    24→                const { execSync } = await import('child_process');
    25→                let globalRoot = '';
    26→                // Detect runtime environment: check Bun global object or process.versions.bun
    27→                const isBun = (typeof globalThis.Bun !== 'undefined') ||
    28→                    (typeof process !== 'undefined' && process.versions.bun);
    29→                if (isBun) {
    30→                    // Bun environment: try multiple ways to find global package path
    31→                    const homeDir = os.homedir();
    32→                    const possiblePaths = [
    33→                        path.join(homeDir, '.bun', 'install', 'global'),
    34→                        path.join(homeDir, '.bun', 'bin', 'node_modules'),
    35→                        '/usr/local/lib/bun/install/global',
    36→                    ];
    37→                    // Try using bun pm cache command to get cache path
    38→                    try {
    39→                        const bunCache = execSync('bun pm cache', { encoding: 'utf-8' }).trim();
    40→                        possiblePaths.unshift(path.join(bunCache, 'global'));
    41→                    }
    42→                    catch {
    43→                        // Ignore error, continue with other paths
    44→                    }
    45→                    // Try each possible path
    46→                    let found = false;
    47→                    for (const possiblePath of possiblePaths) {
    48→                        try {
    49→                            const testPath = path.join(possiblePath, 'z-ai-web-dev-sdk', 'dist', 'index.js');
    50→                            await fs.access(testPath);
    51→                            globalRoot = possiblePath;
    52→                            found = true;
    53→                            break;
    54→                        }
    55→                        catch {
    56→                            // Continue trying next path
    57→                        }
    58→                    }
    59→                    // If all paths failed, use the first default path
    60→                    if (!found) {
    61→                        globalRoot = possiblePaths[0];
    62→                    }
    63→                }
    64→                else {
    65→                    // Node.js environment: use npm root -g
    66→                    globalRoot = execSync('npm root -g', { encoding: 'utf-8' }).trim();
    67→                }
    68→                // Ensure globalRoot is assigned
    69→                if (!globalRoot) {
    70→                    throw new Error('Unable to determine global package path');
    71→                }
    72→                const globalPackagePath = path.join(globalRoot, 'z-ai-web-dev-sdk', 'dist', 'index.js');
    73→                const { default: ZAI } = await import(globalPackagePath);
    74→                return ZAI;
    75→            }
    76→            catch (error3) {
    77→                // Finally fallback to relative path import (for development)
    78→                const __filename = fileURLToPath(import.meta.url);
    79→                const __dirname = path.dirname(__filename);
    80→                const indexPath = path.resolve(__dirname, './index.js');
    81→                const { default: ZAI } = await import(indexPath);
    82→                return ZAI;
    83→            }
    84→        }
    85→    }
    86→}
    87→// Generic parameter parsing function
    88→function parseKeyValueArgs(args) {
    89→    const result = {};
    90→    for (let i = 0; i < args.length; i++) {
    91→        const arg = args[i];
    92→        if (arg.startsWith('--')) {
    93→            const key = arg.slice(2);
    94→            const nextArg = args[i + 1];
    95→            if (nextArg && !nextArg.startsWith('--')) {
    96→                result[key] = nextArg;
    97→                i++;
    98→            }
    99→            else {
   100→                result[key] = 'true';
   101→            }
   102→        }
   103→        else if (arg.startsWith('-') && arg.length === 2) {
   104→            const key = arg.slice(1);
   105→            const nextArg = args[i + 1];
   106→            if (nextArg && !nextArg.startsWith('-')) {
   107→                result[key] = nextArg;
   108→                i++;
   109→            }
   110→            else {
   111→                result[key] = 'true';
   112→            }
   113→        }
   114→    }
   115→    return result;
   116→}

   368→async function saveFile(content, outputPath) {
   369→    try {
   370→        const outputDir = path.dirname(outputPath);
   371→        await fs.mkdir(outputDir, { recursive: true });
   372→        if (Buffer.isBuffer(content)) {
   373→            await fs.writeFile(outputPath, content);
   374→        }
   375→        else {
   376→            await fs.writeFile(outputPath, content, 'utf-8');
   377→        }
   378→        console.log(`✅ File saved to: ${outputPath}`);
   379→    }
   380→    catch (error) {
   381→        throw new Error(`Failed to save file: ${error}`);
   382→    }
   383→}
   384→async function readFileAsBase64(filePath) {
   385→    try {
   386→        const buffer = await fs.readFile(filePath);
   387→        return buffer.toString('base64');
   388→    }
   389→    catch (error) {
   390→        throw new Error(`Failed to read file: ${error}`);
   391→    }
   392→}
   393→// Handle streaming response (SSE format)
   394→async function handleStreamResponse(stream, outputPath) {
   395→    const reader = stream.getReader();
   396→    const decoder = new TextDecoder();
   397→    let buffer = '';
   398→    let fullContent = '';
   399→    const outputLines = [];
   400→    try {
   401→        while (true) {
   402→            const { done, value } = await reader.read();
   403→            if (done)
   404→                break;
   405→            buffer += decoder.decode(value, { stream: true });
   406→            const lines = buffer.split('\n');
   407→            buffer = lines.pop() || ''; // Keep incomplete line
   408→            for (const line of lines) {
   409→                if (line.startsWith('data: ')) {
   410→                    const data = line.slice(6).trim();
   411→                    if (data === '' || data === '[DONE]')
   412→                        continue;
   413→                    try {
   414→                        const json = JSON.parse(data);
   415→                        // Process SSE data
   416→                        if (json.choices && json.choices[0]) {
   417→                            const choice = json.choices[0];
   418→                            if (choice.delta && choice.delta.content) {
   419→                                const content = choice.delta.content;
   420→                                process.stdout.write(content);
   421→                                fullContent += content;
   422→                                outputLines.push(JSON.stringify(json, null, 2));
   423→                            }
   424→                        }
   425→                        else {
   426→                            // Output JSON directly
   427→                            outputLines.push(JSON.stringify(json, null, 2));
   428→                            console.log(JSON.stringify(json, null, 2));
   429→                        }
   430→                    }
   431→                    catch (e) {
   432→                        // Ignore JSON parsing errors
   433→                    }
   434→                }
   435→            }
   436→        }
   437→        // Output newline
   438→        console.log('');
   439→        // If output file is specified, save full content
   440→        if (outputPath) {
   441→            if (fullContent) {
   442→                await saveFile(fullContent, outputPath);
   443→            }
   444→            else if (outputLines.length > 0) {
   445→                await saveFile(outputLines.join('\n'), outputPath);
   446→            }
   447→        }
   448→    }
   449→    catch (error) {
   450→        console.error('❌ Failed to process streaming response:', error);
   451→        throw error;
   452→    }
   453→    finally {
   454→        reader.releaseLock();
   455→    }
   456→}
   457→// Handle chat completion command
   458→async function handleChat(args, client) {
   459→    const params = parseKeyValueArgs(args);
   460→    if (params.help || params.h) {
   461→        showChatHelp();
   462→        return;
   463→    }
   464→    const prompt = params.prompt || params.p;
   465→    if (!prompt) {
   466→        console.error('❌ Error: Missing required parameter --prompt');
   467→        showChatHelp();
   468→        process.exit(1);
   469→    }
   470→    const messages = [];
   471→    if (params.system || params.s) {
   472→        messages.push({
   473→            role: 'system',
   474→            content: params.system || params.s,
   475→        });
   476→    }
   477→    messages.push({
   478→        role: 'user',
   479→        content: prompt,
   480→    });
   481→    const body = {
   482→        messages,
   483→    };
   484→    // Handle thinking parameter (boolean flag, enabled if provided, default disabled)
   485→    if (params.thinking || params.t) {
   486→        body.thinking = { type: 'enabled' };
   487→    }
   488→    else {
   489→        body.thinking = { type: 'disabled' };
   490→    }
   491→    if (params.stream) {
   492→        body.stream = true;
   493→    }
   494→    console.log('🚀 Sending chat request...');
   495→    const response = await client.chat.completions.create(body);
   496→    // Handle streaming response
   497→    if (body.stream && response && typeof response.getReader === 'function') {
   498→        await handleStreamResponse(response, params.output || params.o);
   499→        return;
   500→    }
   501→    const output = params.output || params.o;
   502→    if (output) {
   503→        await saveFile(JSON.stringify(response, null, 2), output);
   504→    }
   505→    else {
   506→        console.log(JSON.stringify(response, null, 2));
   507→    }
   508→}
   509→// Handle vision chat command
   510→async function handleVision(args, client) {
   511→    const params = parseKeyValueArgs(args);
   512→    if (params.help || params.h) {
   513→        showVisionHelp();
   514→        return;
   515→    }
   516→    const prompt = params.prompt || params.p;
   517→    if (!prompt) {
   518→        console.error('❌ Error: Missing required parameter --prompt');
   519→        showVisionHelp();
   520→        process.exit(1);
   521→    }
   522→    const content = [{ type: 'text', text: prompt }];
   523→    // Handle image parameters (possibly multiple) - extract from original args
   524→    for (let i = 0; i < args.length; i++) {
   525→        const arg = args[i];
   526→        if (arg === '--image' || arg === '-i') {
   527→            const imageArg = args[i + 1];
   528→            if (imageArg && !imageArg.startsWith('-')) {
   529→                let imageUrl = imageArg;
   530→                // If it's a local file path, need to convert to base64
   531→                if (!imageArg.startsWith('http://') && !imageArg.startsWith('https://') && !imageArg.startsWith('data:')) {
   532→                    // Local file, read as base64
   533→                    try {
   534→                        const base64 = await readFileAsBase64(imageArg);
   535→                        // Determine MIME type based on file extension
   536→                        const ext = path.extname(imageArg).toLowerCase();
   537→                        const mimeTypes = {
   538→                            '.jpg': 'image/jpeg',
   539→                            '.jpeg': 'image/jpeg',
   540→                            '.png': 'image/png',
   541→                            '.gif': 'image/gif',
   542→                            '.webp': 'image/webp',
   543→                        };
   544→                        const mimeType = mimeTypes[ext] || 'image/jpeg';
   545→                        imageUrl = `data:${mimeType};base64,${base64}`;
   546→                    }
   547→                    catch (error) {
   548→                        console.error(`❌ Unable to read image file: ${imageArg}`);
   549→                        process.exit(1);
   550→                    }
   551→                }
   552→                content.push({
   553→                    type: 'image_url',
   554→                    image_url: { url: imageUrl },
   555→                });
   556→                i++; // Skip processed image parameter value
   557→            }
   558→        }
   559→    }
   560→    const body = {
   561→        messages: [
   562→            {
   563→                role: 'user',
   564→                content,
   565→            },
   566→        ],
   567→    };
   568→    // Handle thinking parameter (boolean flag, enabled if provided, default disabled)
   569→    if (params.thinking || params.t) {
   570→        body.thinking = { type: 'enabled' };
   571→    }
   572→    else {
   573→        body.thinking = { type: 'disabled' };
   574→    }
   575→    if (params.stream) {
   576→        body.stream = true;
   577→    }
   578→    console.log('🚀 Sending vision chat request...');
   579→    const response = await client.chat.completions.createVision(body);
   580→    // Handle streaming response
   581→    if (body.stream && response && typeof response.getReader === 'function') {
   582→        await handleStreamResponse(response, params.output || params.o);
   583→        return;
   584→    }
   585→    const output = params.output || params.o;
   586→    if (output) {
   587→        await saveFile(JSON.stringify(response, null, 2), output);
